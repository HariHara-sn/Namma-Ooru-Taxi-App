groups -> to store groups -- groups can contain X number of users with different roles and X number of devices

    {
        _id,
        name, --group name
        description -- group description 
        groupAdmin, -- Id of the admin user who created this group
        users: [  -- array of user ids who belongs to this group including admin
            {
                userId, 
                role -- roleId -- id of role in userGroupRoles collection
                accepted --whether the added user accepted the invitation,
                addedOn -- time which this user is added
                updatedOn
            }
        ],
        devices: [ -- array of device IDS which the group members can view and track
            {
                deviceId
            }
        ],
        geofences: [ -- array of geofences which is added to the group
            geofenceId
        ]
    }

userGroupRoles -- predefined user roles who are within a group

    {
        _id,
        roleName: admin/read_only/
    }
